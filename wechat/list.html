# 1.定义和使用 behaviors

每个 behavior 可以包含一组属性、数据、生命周期函数和方法，组件引用它时，它的属性、数据和方法会被合并到组件中，生命周期函数也会在对应时机被调用。每个组件可以引用多个 behavior 。 behavior 也可以引用其他 behavior 。(类似vue中 mixins)

创建 index-beh.js

let common = Behavior({
  properties:{
    title:{
      type:String,
      value:"this is Behavior"
    }
  },
  attached: function (options) {

    console.log('componet behaviors attached.')
  },
})
export{common}

// components/Likes/index.js
import { common } from '../../pages/preview/preview-beh'
Component({
  behaviors: [common],
  properties: {
    like: Boolean,
    count: {
      type: Number,
      value: 0
    },
    readOnly: Boolean
  },
  /**
   * 页面的初始数据
   */
  data: {
    yes_url: 'images/like.png',
    no_url: 'images/like@dis.png'
  },
  methods: {
    fromFather() {
      console.log('数据----')
    },
    onLike: function (event) {
      if (this.properties.readOnly) {
        return
      }
      let count = this.properties.count
      count = this.properties.like ? count - 1 : count + 1
      this.setData({
        count: count,
        like: !this.properties.like
      })
      let behavior = this.properties.like ? 'like' : 'cancel'
    }
  },

  /**
   * 生命周期函数--监听页面加载
   */
  attached: function (options) {

    console.log('behaviors attached.')
  }
})

子组件中:

// components/Likes/index.js
Component({
// properties 接受父组件参数
  properties: {
    like: Boolean,
    count: {
      type: Number,
      value: 0,
      observer: "_ober"
    },
    readOnly: Boolean
  },
  /**
   * 页面的初始数据
   */
  data: {
    yes_url: 'images/like.png',
    no_url: 'images/like@dis.png'
  },
  methods: {
    fromFather() {
      console.log('数据----')
    },
    _ober(newVal, oldVal) {
      wx.setStorageSync('count', newVal); // 存储storage
      console.log('over', newVal, oldVal) //数据变化调用
    },
// 子组件事件方法
    onLike(event) {
      if (this.properties.readOnly) {
        return
      }
      let count = this.properties.count
      count = this.properties.like ? count - 1 : count + 1
      this.setData({
        count: count,
        like: !this.properties.like
      })
      let behavior = this.properties.like ? 'like' : 'cancel'
      // 调用 父组件likes 方法
      this.triggerEvent('likes', {
        behavior: behavior  // 传递的参数
      }, {})
    }
  }
})

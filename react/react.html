# React 配置

# 1.按需加载配置

  1. npm run eject 将原本creat react app对webpack，babel等相关配置的封装显示出来
  2. package.json
"babel": {
    "presets": [
      [ "react-app",
        {
          "runtime": "automatic"   // React not define
        }
      ]
    ],
    "plugins": [
      [
        "import",
        {
          "libraryName": "antd",  // antd css 按需加载
          "libraryDirectory": "es",
          "style": "css"
        }
      ]
    ]
  },

# 2.开发环境代理配置

  1. npm install http-proxy-middleware -D http-proxy-middleware 代理安装包
  2. src目录下新建 setupProxy.js 文件(文件名必须是 setupProxy.js)
const {createProxyMiddleware} = require("http-proxy-middleware");
module.exports = function(app){
    app.use(
        createProxyMiddleware("/api",{  // 第一个参数 请求查找path
            target:"http://localhost:3001",  // 真实的请求 origin
            changeOrigin:true,  // 是否改变
            pathRewrite:{
                // "^/api":""  // 路由替换
            }
        })
    )
}

<模块文件名称>/index.js中

import actions from './action';

export default {
  namespaced: true,

store.js中

    import fileName from './filename/index';

    const store = new Vuex.Store({
      modules: {
        fileName: fileName
      },
      state: {
        list: []
      },
      mutations: {
        setList(state, list) {
          state.list = Array.isArray(list) ? list : [];
        }
      },
      // `npm install vuex-persistedstate --save`  Vuex持久化插件(vuex-persistedstate)解决刷新数据消失的问题 根据需求是否引入plugins
      plugins: [
        // 默认是localStorage存储
        createPersistedState({
          paths: ['test']
        }),
        // 可以往sessionStorage存储
        createPersistedState({
          storage: window.sessionStorage,
          paths: ['test']
        })
      ]
    });

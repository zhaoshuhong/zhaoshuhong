# vue axios

1.安装axios npm install axios --save

2.在src目录下创建axios-config.js文件(配置文件)，内容如下：

// 引入axios依赖
// 引入axios依赖
import axios from 'axios';

const http = axios.create({
  // baseURL: Vue.prototype.$baseURL,
  headers: {
    'Content-type': 'application/json'
  }
});

const httpErrorHandler = (status, other) => {
  // 判断错误状态码
  switch (status) {
  case 401:
    break;
  case 403:
    break;
  case 404:
    break;
  default:
    console.log(other);
  }
};
// 目前处于pending状态的请求列表
const pending = [];
/**
 * 取消请求
 * @param {*} ever
 */
const removePending = (ever) => {
  // 遍历所有处于pending状态的请求
  for (const p in pending) {
    // 判断当前请求是否有相同请求处于pending状态
    if (pending[p].u === `${ever.url}&${ever.method}`) {
      // 当前请求有相同请求处于pending状态，取消pending状态的该请求
      pending[p].f(ever);
      pending.splice(p, 1);
    }
  }
};

http.interceptors.request.use(
  (config) => {
    // 发送请求前，把相同的请求从pending列表中移除
    removePending(config);
    config.cancelToken = new axios.CancelToken((c) => {
      // 将请求加入pending列表
      pending.push({ u: `${config.url}&${config.method}`, f: c });
    });
    return config;
  },
  (error) => Promise.reject(error)
);

http.interceptors.response.use(
  (res) => {
    // 对于已经完成的请求，从pending列表中移除
    removePending(res.config);
    if (res.status === 200) {
      return Promise.resolve(res);
    }
    return Promise.reject(res);
  },
  (error) => {
    const { response } = error;
    if (axios.isCancel(error)) {
      console.log('Rquest canceled', error.message); // 请求如果被取消，这里是返回取消的message
    }
    if (response) {
      httpErrorHandler(response.status, response.data.message);
    }
    return Promise.reject(response);
  }
);
export default http;

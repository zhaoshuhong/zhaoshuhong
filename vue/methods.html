# 1.监听生命周期函数 hook

  • 在Vue组件中，可以用过$on,$once去监听所有的生命周期钩子函数，如监听组件的钩子函数可以写成 this.$on('hook:[hookname]', () => {})
export default {
  mounted() {
    // 监听窗口发生变化，resize组件
    window.addEventListener('resize', this.$_init)
    // 通过hook监听组件销毁钩子函数，并取消监听事件
    this.$once('hook:beforeDestroy', () => {
      window.removeEventListener('resize', this.$_init)
    })
  },
  methods: {
    $_init() {
      // resize init
    }
  }
}

# 2.组件以api引入(extend)

1.新建一个简单的组件 hello.vue

<template>
  <div>{{text}}</div><div @click="$_destroy">销毁</div>

export default {
  name: 'hello',
  data () {
    return {
      text: ''
    }
  },
  methods: {
    $_destroy() {

      this.$destroy() //销毁
    }
  }
}
</script>

2.创建hello.js

import Vue from 'vue';
import HelloTemplate from './hello.vue';

// 使用extend方法创建vue的子类
const HelloConstructor = Vue.extend(HelloTemplate);
// 添加销毁方法 close
HelloConstructor.prototype.close = function() {
  if (this.$el && this.$el.parentNode) {
    this.$el.parentNode.removeChild(this.$el)
  }
  // 调用组件的$destroy方法进行组件销毁
  this.$destroy()
}


// 使用这个方法调用hello组件
function Hello(options) {
  options = options || {};
  if (typeof options === 'string') {
    options = {
      text: options
    }
  }

  // 实例化子组件，然后获取到DOM结构并挂载到body上
  const helloInstence = new HelloConstructor({data: options});
  helloInstence.vm = helloInstence.$mount();
  document.body.appendChild(helloInstence.vm.$el);
}
export default Hello;

export default function WithConsole(WrappedComponent) {
  return {
    mounted() {
      console.log('I have already mounted');
    },
    props: WrappedComponent.props,
    render(h) {
      const slots = Object.keys(this.$slots)
        .reduce((arr, key) => arr.concat(this.$slots[key]), [])
        // 手动更正 context
        .map((vnode) => {
          vnode.context = this._self;
          return vnode;
        });

      return h(WrappedComponent, {
        on: this.$listeners,
        props: this.$props,
        attrs: this.$attrs
      }, slots);
    }
  };
}

function withTrim(Ctor) {
  return {
    functional: true,
    render(h, context) {
      return h(Ctor, {
        ...context.data,
        props: context.props,
        slots: context.slots(),
        scopedSlots: context.scopedSlots,
        on: {
          ...context.listeners,
          input(value) {
            if (context.listeners.input) {
              return context.listeners.input(value.trim());
            }
          }
        }
      });
    }
  };
}

# 4.JSX方式

<script type="text/jsx">
import moment from 'moment';

export default {
  name: 'CSTableIndexJSX',
  props: {
    tableData: {
      type: Array,
      default: () => []
    }
  },
  data() {
    return {
      columnList: [
        {
          prop: 'certificateNumber',
          label: '编号'
        },
        {
          prop: 'productTypeName',
          label: '类型'
        },
        {
          prop: 'certificateMaker',
          label: '持证人姓名'
        },
        {
          prop: 'productModelName',
          label: '型号'
        },
        {
          prop: 'productModelTypeName',
          label: '型别'
        },
        {
          prop: 'approvalDate',
          label: '颁证日期',
          render: scope => <p
            class="cell"
            style="white-space: nowrap"
          >
            {
              this.timeFilter(scope.row.approvalDate)
            }
          </p>
        }
      ]
    };
  },
  methods: {
    // 时间过滤
    timeFilter(value) {
      return moment(value)
        .format('YYYY-MM-DD');
    }
  },
  render(h) {
    return (
      <div style="height:100%">
        <el-table
          data={this.tableData}
          header-row-class-name="table-show"
          stripe
          style="width: 100%"
          height="100%"
        >
          {
            this.columnList.map(columns => <el-table-column
              prop={columns.prop}
              label={columns.label}
              show-overflow-tooltip
              {...{
                scopedSlots: {
                  default: scope => (Object.prototype.hasOwnProperty.call(columns, 'render') ? columns.render(scope) : scope.row[columns.prop])
                }
              }}
            >
            </el-table-column>)
          }

        </el-table>
      </div>
    );
  }
};
</script>

# 6.vue自定义指令 --directive

import Vue from 'vue'
import LoadingComponent from './loading'
// 使用 Vue.extend构造组件子类
const LoadingContructor = Vue.extend(LoadingComponent)

// 定义一个名为loading的指令
Vue.directive('loading', {
  /**
   * 只调用一次，在指令第一次绑定到元素时调用，可以在这里做一些初始化的设置
   * @param {*} el 指令要绑定的元素
   * @param {*} binding 指令传入的信息，包括 {name:'指令名称', value: '指令绑定的值',arg: '指令参数 v-bind:text 对应 text'}
   */
  bind(el, binding) {
    const instance = new LoadingContructor({
      el: document.createElement('div'),
      data: {}
    })
    el.appendChild(instance.$el)
    el.instance = instance
    Vue.nextTick(() => {
      el.instance.visible = binding.value
    })
  },
  /**
   * 所在组件的 VNode 更新时调用
   * @param {*} el
   * @param {*} binding
   */
  update(el, binding) {
    // 通过对比值的变化判断loading是否显示
    if (binding.oldValue !== binding.value) {
      el.instance.visible = binding.value
    }
  },
  /**
   * 只调用一次，在 指令与元素解绑时调用
   * @param {*} el
   */
  unbind(el) {
    const mask = el.instance.$el
    if (mask.parentNode) {
      mask.parentNode.removeChild(mask)
    }
    el.instance.$destroy()
    el.instance = undefined
  }
})

# 7.mixins 合并策略更改

Vue.mixin({
  created() {
    this.messages.push('global mixin hook called');
  }
});

const mixin = {
  created() {
    this.messages.push('mixin hook called');
  },
};
// 通过 Vue.config.optionMergeStrategies.name 来判断mixins components name 来进行合并

const mergeCreatedStrategy = Vue.config.optionMergeStrategies.created;
Vue.config.optionMergeStrategies.created = function strategy(toVal, fromVal) {
  if (fromVal.overrideOthers) {
    // Selectively override hooks injected from mixins
    return [fromVal];
  }
  return mergeCreatedStrategy(toVal, fromVal);
};

const app = {
  el: '#app',
  mixins: [mixin],
  data: { messages: [] },
  created() {
    this.messages.push('component hook called');
  },
};
// Comment or change this line to control whether the mixin created hook is applied
app.created.overrideOthers = true;

new Vue(app);

# 8.vue.observable替代vux

如果不打算开发大型单页应用，使用 Vuex 可能是繁琐冗余的。这时候我们就可以使用Vue2.6提供的新API Vue.observable手动打造一个Vuex
1.创建store.js

import Vue from 'vue'

// 通过Vue.observable创建一个可响应的对象
export const store = Vue.observable({
  userInfo: {}
})

// 定义 mutations, 修改属性
export const mutations = {
  setUserInfo(userInfo) {
    store.userInfo = userInfo
  }
}

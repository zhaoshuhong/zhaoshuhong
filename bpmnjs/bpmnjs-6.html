# bpmn.js 学习使用总结(bpmnjs-中文翻译)

# 操作

  • 新建 config > customTranslate.js

  • 在vue页面中引入

<script >
import customTranslate from './config/customTranslate'; // 中文

// 实例化 在 additionalModules中注入
this.modeler = new Modeler({
  container: this.$refs.canvas,
  linting: {
    active: true, // 是否在实例化之后就开启流程校验
    bpmnlint: bpmnlintConfig
  },
  additionalModules: [
    customProvider,
    minimapModule,
    lintModule,
    {
      translate: ['value', customTranslate] // 引入
    }
  ],
  moddleExtensions: {
    flowable: flowableModdle
  }
});
</script>

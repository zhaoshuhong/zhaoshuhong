# bpmn.js 元素渲染Renderer

我们通过自定义 Palette组件改变了元素栏样式，但在拖拽和画布渲染中要相应更改样式。本次只在在默认的Renderer基础上修改

# bpmn-js 配置Renderer组件

  • 新建config 文件夹来存储配置文件 在里面新建render > customRenderer.js 从bpmn-jsnode_moudles 中复制
  • 创建util.js 用来放哪些类型是需要我们自定义的, 从而在渲染的时候就可以与不需要自定义的元素作区分。

export default class CustomRenderer extends BaseRenderer { // 继承BaseRenderer
  constructor(eventBus, bpmnRenderer) {
    super(eventBus, HIGH_PRIORITY)
    this.bpmnRenderer = bpmnRenderer
  }

  canRender(element) {
    // ignore labels
    return !element.labelTarget
  }

  drawShape(parentNode, element) { // 核心函数就是绘制shape
    const {
      type,
      businessObject
    } = element // 获取到类型
    if(customElements.includes(type)) {
      if (type !== 'label') {
        let types = type;
        const typesArr = businessObject.eventDefinitions;
        if(typesArr && typesArr.length) {
          types = typesArr[0].$type;
        }
        if (customElements.includes(types)) { // or customConfig[type]
          const {
            url,
            attr,
            color,
            className
          } = customConfig[types]
          //可以设置节点默认值
          // if (!element.businessObject.name && defaultName) {
          //     element.businessObject.name = defaultName
          // }

          const customIcon = svgCreate('use', {
            ...attr,
            href: url,
            class: className,
            fill: color,
          })
          element['width'] = attr.width //  直接修改了元素的宽高
          element['height'] = attr.height
          svgAppend(parentNode, customIcon)
          // 判断是否有name属性来决定是否要渲染出label
          // if (!hasLabelElements.includes(type) && element.businessObject.name) {
          //当类型为UserTask时,自定义绘制label标签（标签会在shape里面）
          if (type === 'bpmn:UserTask' && element.businessObject.name) {
            const textWidth = element.businessObject.name.pxWidth('normal 13px')
            const text = svgCreate('text', {
              x: attr.x + (attr.width / 2 - textWidth / 1.6),
              y: attr.y + attr.height + 15,
              fontSize: "13px",
              fill: "#000"
            })
            text.innerHTML = element.businessObject.name
            svgAppend(parentNode, text)
          }
          // renderLabel(parentNode, element.label)

          return customIcon
        }

      }
    }
    const shape = this.bpmnRenderer.drawShape(parentNode, element);
    this.setShapeProperties(shape, element)
    return shape
  }

  setShapeProperties (shape, element) {
    const type = element.type // 获取到的类型
    if (propertiesConfig[type]) {
      const properties = propertiesConfig[type]
      Object.keys(properties).forEach(prop => {
        shape.style.setProperty(prop, properties[prop])
      })
    }
  }

  getShapePath(shape) {
    return this.bpmnRenderer.getShapePath(shape)
  }
}

CustomRenderer.$inject = ['eventBus', 'bpmnRenderer']

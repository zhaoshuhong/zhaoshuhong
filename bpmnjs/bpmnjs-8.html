<template>
  <div class="bpmn-content" v-loading="pageLoading">
    <div class="bpmn-content-header">
      <div>
        <el-button-group style="margin-right: 20px">
          <el-tooltip effect="light" content="放大" placement="top-start">
            <el-button icon="el-icon-plus" @click="zoomViewport"></el-button>
          </el-tooltip>
          <el-tooltip effect="light" content="自适应居中显示" placement="top-start">
            <el-button @click="fixDesktop">{{ zoomLabel }}</el-button>
          </el-tooltip>
          <el-tooltip effect="light" content="缩小" placement="top-start">
            <el-button icon="el-icon-minus" @click="zoomViewport(false)"></el-button>
          </el-tooltip>
        </el-button-group>

        <el-button-group v-if="!disabled">
          <el-tooltip effect="light" content="撤销" placement="top-start">
            <el-button icon="el-icon-refresh-left" @click="commandStackChange('undo')"></el-button>
          </el-tooltip>
          <el-tooltip effect="light" content="恢复" placement="top-start">
            <el-button icon="el-icon-refresh-right" @click="commandStackChange('redo')"></el-button>
          </el-tooltip>
          <el-tooltip effect="light" content="清除" placement="top-start">
            <el-button type="danger" icon="el-icon-delete" @click="clearData"></el-button>
          </el-tooltip>
        </el-button-group>
      </div>

      <el-button-group>
        <el-button type="info" @click="previewXml">预览xml</el-button>
        <!--        <el-button @click="saveForm">保存(-临时-)</el-button>-->
        <el-button v-if="!disabled" type="primary" @click="saveEditor">暂存</el-button>
        <el-button v-if="!disabled" type="warning" @click="deployedBpmn">部署</el-button>
      </el-button-group>

    </div>
    <div class="bpmn-content-box">
      <div v-if="!disabled" class="content-tip">注意：第一个用户任务必须为发起人，办理人需设置为${_INITATOR_}</div>
      <div ref="canvas" class="bpmn-content-box-canvas"/>
      <bpmn-panel
        v-if="!disabled"
        :key="renderKey"
        :form-data="formData"
        :modeler="modeler"
        :element="element"></bpmn-panel>
    </div>
    <event-code-eidt-dialog
      :cur-data="curData"
      :visible.sync="isShowXmlEditor">
    </event-code-eidt-dialog>
  </div>
</template>

<script>
import Modeler from 'bpmn-js/lib/Modeler';
import BpmnViewer from 'bpmn-js/lib/Viewer';
import ModelingModule from 'bpmn-js/lib/features/modeling';
import MoveCanvasModule from 'diagram-js/lib/navigation/movecanvas';
import minimapModule from 'diagram-js-minimap';
import lintModule from 'bpmn-js-bpmnlint';
import { v4 as uuidv4 } from 'uuid';
import * as bpmnlintConfig from './config/lint/packed-config';
import customProvider from './config/index';
import customTranslate from './config/customTranslate'; // 中文
import flowableModdle from './config/flowable.json'; // bpmn 属性property 用于面板修改属性
import BpmData from './config/paletteData';
import BpmnPanel from './Panel.vue';
import EventCodeEidtDialog from './EventCodeEidtDialog.vue';
import {
  saveXMLApi, deployApi, saveModelEditorApi, queryByIdApi
} from '../../api';

export default {
  name: 'BpmnContent',
  components: {
    BpmnPanel,
    EventCodeEidtDialog
  },
  props: {
    formId: {
      type: String,
      default: ''
    },
    disabled: {
      type: Boolean,
      default: false
    }
  },
  data() {
    return {
      pageLoading: false, // 页面显示loading
      // formId: '67aba6a0-b27a-11ed-a33f-00ff142f26de', // 表单id
      renderKey: 0, // panel渲染更新的key
      curData: '', // xml内容
      isShowXmlEditor: false, // xml 预览弹窗
      zoom: 1, // bpmn 缩放尺寸
      rendering: true, // xml 渲染状态
      xml: null, // 渲染得到xml
      modeler: null, // bpmn 实例
      element: null, // 当前点击ele
      // bpmn properties panel数据
      formData: {
        deployed: false, // 是否已经部署
        processCategory: '', // 流程分类
        id: '', // 流程id
        name: '', // 流程名称
        documentation: '', // 流程描述
        eventLister: [],
        lcbl: [] // 流程变量
      }
    };
  },
  provide() {
    return {
      pageLoading: (val) => {
        this.pageLoading = val;
      } // 页面loading
    };
  },
  computed: {
    // 缩放尺寸
    zoomLabel() {
      return `${Math.floor(this.zoom * 100)}%`;
    },
    // bpmn 是否render完成
    isHadRender() {
      return this.modeler && !this.rendering;
    }
  },
  mounted() {
    if (!this.formId) {
      this.$message.error('流程Key不存在');
      // return;
    }
    // 工作流初始化
    this.diagramsInit();
  },
  methods: {
    // bpmn 撤销/恢复
    commandStackChange(type) {
      if (this.modeler) {
        this.modeler.get('commandStack')[type]();
      }
    },
    // 清空
    clearData() {
      if (!this.modeler) {
        return;
      }
      const documentation = this.formData.documentation || '';
      const id = this.formData.id || `process_${uuidv4()}`;
      const name = this.formData.name || `name_${uuidv4()}`;
      const processCategory = this.formData.processCategory || '';
      this.xml = `<?xml version="1.0" encoding="UTF-8"?>
<definitions xmlns="http://www.omg.org/spec/BPMN/20100524/MODEL" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI" xmlns:flowable="http://flowable.org/bpmn" xmlns:xsd="http://www.w3.org/2001/XMLSchema" targetNamespace="http://www.flowable.org/processdef">
  <process id="${id}" name="${name}" flowable:processCategory="${processCategory}">
    <documentation>${documentation}</documentation>
  </process>
  <bpmndi:BPMNDiagram id="BPMNDiagram_flow">
    <bpmndi:BPMNPlane id="BPMNPlane_flow" bpmnElement="${id}" />
  </bpmndi:BPMNDiagram>
</definitions>`;
      this.createNewDiagrams(this.xml);
    },
    // 预览 xml数据
    async previewXml() {
      try {
        const { xml } = await this.modeler.saveXML({ format: true });
        this.curData = xml;
        this.isShowXmlEditor = true;
      } catch (err) {
        console.log(err);
      }
    },
    // 保存数据 临时测试生成表单id
    async saveForm() {
      try {
        const { xml } = await this.modeler.saveXML({ format: true });
        this.isShowXmlEditor = true;
        this.curData = xml;
        const params = {
          key: this.formData.id,
          name: this.formData.name,
          category: this.formData.processCategory,
          description: this.formData.documentation
        };
        this.pageLoading = true;
        saveXMLApi(params)
          .then((res) => {
            console.log(res);
            this.formId = res.data.content;
          })
          .catch((err) => {
            console.log(err);
          })
          .finally(() => {
            this.pageLoading = false;
          });
      } catch (err) {
        console.log(err);
      }
    },
    // 保存editor
    async saveEditor() {
      try {
        this.pageLoading = true;
        const { xml } = await this.modeler.saveXML({ format: true });
        this.curData = xml;
        const params = {
          id: this.formId,
          editor: xml
        };
        saveModelEditorApi(params)
          .then((r) => {
            this.$message.success('暂存成功');
            console.log(r);
          })
          .catch((errs) => {
            console.log(errs);
          })
          .finally(() => {
            this.pageLoading = false;
          });
      } catch (err) {
        console.log(err);
      }
    },
    // 部署
    async deployedBpmn() {
      this.pageLoading = true;
      try {
        const { xml } = await this.modeler.saveXML({ format: true });
        this.curData = xml;
        const params = {
          id: this.formId,
          editor: xml
        };
        await saveModelEditorApi(params)
          .then((r) => {
            console.log(r);
          })
          .catch((errs) => {
            this.pageLoading = false;
            console.log(errs);
          });
        deployApi({ id: this.formId })
          .then((r) => {
            console.log(r);
            this.$message.success('部署成功');
          })
          .catch((errs) => {
            console.log(errs);
          })
          .finally(() => {
            this.pageLoading = false;
          });
      } catch (err) {
        this.pageLoading = false;
        console.log(err);
      }
    },
    // 获取editor数据信息
    getEditorInfo() {
      this.pageLoading = true;
      queryByIdApi({ id: this.formId })
        .then((res) => {
          console.log(res);
          Object.assign(this.formData, res.data.content);
          if (res.data.content.editor) {
            this.xml = res.data.content.editor;
          } else {
            this.formData.documentation = res.data.content.description || '';
            this.formData.id = res.data.content.id || `process_${uuidv4()}`;
            this.formData.name = res.data.content.name || `name_${uuidv4()}`;
            this.formData.processCategory = res.data.content.category || '';

            this.xml = `<?xml version="1.0" encoding="UTF-8"?>
            <definitions xmlns="http://www.omg.org/spec/BPMN/20100524/MODEL" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI" xmlns:flowable="http://flowable.org/bpmn" xmlns:xsd="http://www.w3.org/2001/XMLSchema" targetNamespace="http://www.flowable.org/processdef">
              <process id="${this.formData.id}" name="${this.formData.name}" flowable:processCategory="${this.formData.processCategory}">
                <documentation>${this.formData.documentation}</documentation>
              </process>
              <bpmndi:BPMNDiagram id="BPMNDiagram_flow">
                <bpmndi:BPMNPlane id="BPMNPlane_flow" bpmnElement="${this.formData.id}" />
              </bpmndi:BPMNDiagram>
            </definitions>`;
          }
          this.createNewDiagrams(this.xml);
        })
        .catch(() => {
        })
        .finally(() => {
          this.pageLoading = false;
        });
    },
    // 缩放尺寸设置
    zoomViewport(zoomIn = true) {
      if (!zoomIn && this.zoom <= 0.1) {
        return;
      }
      this.zoom = this.modeler.get('canvas')
        .zoom();
      this.zoom += (zoomIn ? 0.1 : -0.1);
      this.zoom = Math.round(this.zoom * 10) / 10;
      this.modeler.get('canvas')
        .zoom(this.zoom);
    },
    // bpmn 渲染居中
    fixDesktop() {
      if (this.isHadRender) {
        this.zoom = this.modeler.get('canvas')
          .zoom('fit-viewport');
        const bbox = document.querySelector('.bpmn-content-box .viewport')
          .getBBox();
        const currentViewbox = this.modeler.get('canvas')
          .viewbox();
        const elementMid = {
          x: bbox.x + bbox.width / 2,
          y: bbox.y + bbox.height / 2
        };
        this.modeler.get('canvas')
          .viewbox({
            x: elementMid.x - currentViewbox.width / 2,
            y: elementMid.y - currentViewbox.height / 2,
            width: currentViewbox.width,
            height: currentViewbox.height
          });
        // this.zoom = (bbox.width / currentViewbox.width) * 1.8;
      }
    },
    // modeler事件监听
    handleModeler() {
      this.modeler.on('root.added', (e) => {
        console.log('root:added', e.element);
        if (e.element.type === 'bpmn:Process') {
          this.element = null;
          this.$nextTick()
            .then(() => {
              this.getProcessData(e);
            });
        }
      });
      this.modeler.on('element.click', (e) => {
        const { element } = e;
        console.log('click', element);
        if (element.type === 'bpmn:Process') {
          // this.element = element;
          this.getProcessData(e);
          return;
        }
        if (element.type === 'label') {
          this.element = element.labelTarget;
          this.renderKey += 1;
          return;
        }
        this.$nextTick()
          .then(() => {
            this.element = element;
          });
      });
      this.modeler.on('selection.changed', (e) => {
        // hack 同类型面板不刷新
        this.element = null;
        const element = e.newSelection[0];
        if (element) {
          this.$nextTick()
            .then(() => {
              this.element = element;
            });
        }
      });
      // this.modeler.on('selection.changed', (e) => {
      //   // hack 同类型面板不刷新
      //   this.element = null;
      //   const element = e.newSelection[0];
      //   if (element) {
      //     this.$nextTick().then(() => {
      //       this.element = element;
      //     });
      //   }
      // });
    },
    initEvent() {
      const extensionElements = this.element.businessObject?.extensionElements?.values || [];
      return extensionElements
        .filter((item) => item.$type === 'flowable:ExecutionListener')
        .map((item) => {
          let type;
          if ('class' in item) type = 'class';
          if ('expression' in item) type = 'expression';
          if ('delegateExpression' in item) type = 'delegateExpression';
          return {
            event: item.event,
            type,
            className: item[type],
            params: item.fields?.map((field) => {
              let fieldType;
              if ('stringValue' in field) fieldType = 'stringValue';
              if ('expression' in field) fieldType = 'expression';
              return {
                name: field.name,
                type: fieldType,
                value: field[fieldType]
              };
            }) ?? []
          };
        }) ?? [];
    },
    // bpmn properties
    updateProperties(properties) {
      const modeling = this.modeler.get('modeling');
      modeling.updateProperties(this.element, properties);
    },
    // 获取流程信息
    getProcessData(e) {
      this.element = e.element;
      this.formData.id = this.element.businessObject.id;
      this.formData.name = this.element.businessObject.name || '';
      this.formData.processCategory = this.element.businessObject.$attrs['flowable:processCategory'] || '';
      const documentation = this.element.businessObject.documentation || [];
      this.formData.documentation = documentation.length ? this.element.businessObject.documentation[0].text : '';
      this.formData.eventLister = this.initEvent();
      this.formData.lcbl = [];
      const lcbl = this.element.businessObject.$attrs['flowable:lcbl'] || '';
      const lcblms = this.element.businessObject.$attrs['flowable:lcblms'] || '';
      const lcblArr = lcbl ? lcbl.split(',') : [];
      const lcblmsArr = lcblms ? lcblms.split(',') : [];
      lcblArr.forEach((item, index) => {
        this.formData.lcbl.push({
          key: item,
          value: lcblmsArr[index]
        });
      });
      if ((this.formData.lcbl.length && this.formData.lcbl[0].key !== '_INITATOR_') || this.formData.lcbl.length === 0) {
        this.formData.lcbl.unshift({
          key: '_INITATOR_',
          value: '发起人'
        });
      }
    },
    // bpmn 初始化渲染
    diagramsInit() {
      // 生成实例
      if (!this.modeler) {
        if (!this.disabled) {
          this.modeler = new Modeler({
            container: this.$refs.canvas,
            linting: {
              active: true, // 是否在实例化之后就开启流程校验
              bpmnlint: bpmnlintConfig
            },
            additionalModules: [
              customProvider,
              minimapModule,
              lintModule,
              {
                translate: ['value', customTranslate]
              }
            ],
            moddleExtensions: {
              flowable: flowableModdle
            }
          });
          console.log('modeler:', this.modeler);
          this.renderPalette();
          this.handleModeler();
        } else {
          this.modeler = new BpmnViewer({
            container: this.$refs.canvas,
            additionalModules: [
              ModelingModule, // 基础工具 MoveModule、SetColor 等依赖于此
              MoveCanvasModule // 移动整个画布
            ]
          });
        }
      }
      this.getEditorInfo();
    },
    // bpmn 图render
    async createNewDiagrams(xml) {
      // 将字符串转换成图显示出来 替换xml中的 cdata标记中<
      const data = xml.replace(/<!\[CDATA\[(.+?)]]>/g, (match, str) => str.replace(/</g, '&lt;'));
      this.rendering = true;
      await this.modeler.importXML(data);
      this.rendering = false;
    },
    // bpmn 左侧工具栏样式渲染调整
    renderPalette() {
      console.log('bpmn refs:', this.$refs.canvas);
      try {
        // 修改tools 样式
        const djsPalette = document.querySelector('.djs-palette.open'); // 获取 bpmn 设计器实例
        const djsPalStyle = {
          width: '130px',
          padding: '5px',
          background: '#FFFFFF',
          left: '20px',
          borderRadius: '10px'
        };
        Object.keys(djsPalStyle)
          .forEach((itemKey) => {
            djsPalette.style[itemKey] = djsPalStyle[itemKey];
          });
        const palatteEntryes = document.querySelector('.djs-palette.open > .djs-palette-entries').children;
        for (let j = 0; j < palatteEntryes.length; j += 1) {
          const itemDom = palatteEntryes[j];
          const controlStyle = {
            display: 'flex',
            justifyContent: 'flex-start',
            alignItems: 'center',
            width: '100%',
            padding: '5px',
            boxSizing: 'border-box'
          };
          for (let i = 0; i < itemDom.children.length; i += 1) {
            const controlItem = itemDom.children[i];
            Object.keys(controlStyle)
              .forEach((x) => {
                controlItem.style[x] = controlStyle[x];
              });
            if (controlItem.className
              && controlItem.dataset
              && controlItem.className.indexOf('entry') !== -1) {
              const controlProps = new BpmData().getControl(
                controlItem.dataset.action
              );

              if (controlProps.title) {
                controlItem.innerHTML = `<div style='font-size: 14px;font-weight:500;margin-left:15px;'>${
                  controlProps.title
                }</div>`;
              } else {
                controlItem.style.display = 'none';
              }
            }
          }
        }
      } catch (e) {
        console.log(e);
      }
    }
  }
};
</script>
